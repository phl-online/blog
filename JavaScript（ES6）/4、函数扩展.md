# 函数扩展

## 函数参数的默认值
```js
// 1、默认值的基本用法

// 常规写法
function f(x，y) {
  y = y|| 'world';
  console.log(x, y);
}

log('Hello') // Hello world
// y 没有赋值，默认的就是world

log('Hello', 'China') // Hello China
log('Hello', '') // Hello world
// 例如y赋值了是'',但是对应的布尔值为false 不起作用，所以又是默认值了

// es6
function f(x,y='world') {
   console.log(x, y);
}


log('Hello') // Hello world
log('Hello', 'China') // Hello China
log('Hello', '') // Hello 

// 2、函数默认值的参数不能重复声明

// 情况一、参数的变量是默认声明的，不能重复声明
function f(x=5) {
  const x=3;
}
f() // 报错

// 情况二、使用参数默认值，函数不能有同名参数
function foo(x, x, y = 1) {
  console.log(x, y);
}

f(3,4,5) // 报错

// 情况三、参数默认值的计算是惰性的
// 每次调用都会重新计算参数里的表达式
let x=1;
function f(y=x+1) {
  console.log(y);
}
f(); //2
x=2;
f(); //3

// 3、函数默认值与解构默认值的结合使用
function f({a=0,b=0}={a:3}) {
   console.log(a, b);
}

f({a: 3, b: 4}); //3 4
f({a: 3}); //3 0
f({}); //0 0
f(); //3 0

// 4、实参省略的方法
// 在解析赋值里参数省略的方法
let [,b,c]=[1,2,3]
console.log(a, b); //2,3

// 但是在函数里不能这样省略
function f(x,y=2,z) {
  console.log(x,y)
}
f(,3,4);// 错误
f(undefined,4,3);

```


## 箭头函数

* 基本用法
  * `=>`去定义函数

* 函数

* 实际场景
```js
// 1、一个参数，一个返回
// es6
let f= v=>v;

//等价于
var  f= function (v) {
  return v
}

// 2、没有参数，只有一个返回
//es6
let f=()=>5;

//等价于
var f= function () {
  return 5
}

//3、多个参数，1个返回
//es6
let f=(a,b,c)=>a+b+c;

//等价于
var f= function (a,b,c) {
  return a+b+c; 
}

//4、函数里面有多条语句
//es6
let f2=（a,b）=>{
  let c=0;
  c+=a*a;
  c+=b*b;
  c=Math.sqrt(c);
  return c;
}

//等价于
var f2=function (a,b) {
  let c=0;
  c+=a*a;
  c+=b*b;
  c=Math.sqrt(c);
  return c;
}
















```
